----------------------------------------------------------------------
-- Provable judgements have well-formed contexts
----------------------------------------------------------------------
⊢ok :
  {Γ : Cx}
  {J : Jg}
  (_ : Γ ⊢ J)
  → ---------
  Ok Γ

⊢ok (⊢conv q _) = ⊢ok q
⊢ok (⊢𝐯 q _) = q
⊢ok (⊢𝐔 q) = q
⊢ok (⊢𝚷 _ q _) = ⊢ok q
⊢ok (⊢𝛌 _ _ q _) =  ⊢ok q
⊢ok (⊢∙ _ q _ _ _) = ⊢ok q
⊢ok (⊢𝐄𝐪 q _ _) = ⊢ok q
⊢ok (⊢𝐫𝐞𝐟𝐥 q _) = ⊢ok q
⊢ok (⊢𝐄𝐦𝐩 q) = q
⊢ok (⊢𝐞𝐦𝐩 q _ ) = ⊢ok q
⊢ok (⊢𝐍𝐚𝐭 q) = q
⊢ok (⊢𝐳𝐞𝐫𝐨 q) = q
⊢ok (⊢𝐬𝐮𝐜𝐜 q) = ⊢ok q
⊢ok (⊢𝐧𝐫𝐞𝐜 _ q _ _ _) = ⊢ok q
⊢ok (Refl q) = ⊢ok q
⊢ok (Symm q) = ⊢ok q
⊢ok (Trans q _) = ⊢ok q
⊢ok (＝conv q _) = ⊢ok q
⊢ok (𝚷Cong _ q _ _) = ⊢ok q
⊢ok (𝛌Cong _ q _ _ _) = ⊢ok q
⊢ok (∙Cong _ _ _ q _ _ _) = ⊢ok q
⊢ok (𝐄𝐪Cong q _ _) = ⊢ok q
⊢ok (𝐫𝐞𝐟𝐥Cong q _) = ⊢ok q
⊢ok (𝐞𝐦𝐩Cong q _) = ⊢ok q
⊢ok (𝐬𝐮𝐜𝐜Cong q) = ⊢ok q
⊢ok (𝐧𝐫𝐞𝐜Cong _ _ q _ _ _) = ⊢ok q
⊢ok (𝚷Beta _ _ q _ _) = ⊢ok q
⊢ok (𝐍𝐚𝐭Beta₀ _ q _ _) = ⊢ok q
⊢ok (𝐍𝐚𝐭Beta₊ _ q _ _ _) = ⊢ok q
⊢ok (𝚷Eta _ q _ _ _ _) = ⊢ok q
⊢ok (Reflect q _ _ _) = ⊢ok q
⊢ok (UIP q _ _ _ _) = ⊢ok q

ok＝ :
  {Γ Γ' : Cx}
  (_ : ⊢ Γ ＝ Γ')
  → -------------
  Ok Γ

ok＝ ＝◇ = ok◇
ok＝ (＝⨟ q _ (q' ∉∪ _) h _) = ok⨟ h q' (ok＝ q)

＝ok :
  {Γ Γ' : Cx}
  (_ : ⊢ Γ ＝ Γ')
  → -------------
  Ok Γ'

＝ok ＝◇ = ok◇
＝ok (＝⨟ q _ (_ ∉∪ q') _ h) = ok⨟ h q' (＝ok q)

dom▷ :
  {Δ Γ : Cx}
  (_ : Δ ▷ Γ)
  → -----------
  dom Γ ⊆ dom Δ

dom▷ ▷◇            = id
dom▷ (▷proj q _ _) = ∈∪₁ ∘ dom▷ q
dom▷ (▷⨟ q _ _ _)  = ⊆ub (∈∪₁ ∘ dom▷ q) ∈∪₂

▷Ok :
  {Δ Γ : Cx}
  (_ : Δ ▷ Γ)
  → ---------
  Ok Γ

▷Ok ▷◇              = ok◇
▷Ok (▷proj q _ _)   = ▷Ok q
▷Ok (▷⨟ q₀ q₁ q₂ _) = ok⨟ q₁ (⊆∉ (dom▷ q₀) q₂) (▷Ok q₀)

Ok▷ :
  {Δ Γ : Cx}
  (_ : Δ ▷ Γ)
  → ---------
  Ok Δ

Ok▷ ▷◇               = ok◇
Ok▷ (▷proj q₀ q₁ q₂) = ok⨟ q₁ q₂ (Ok▷ q₀)
Ok▷ (▷⨟ q₀ _ q₂ h)   = ok⨟ h q₂ (Ok▷ q₀)

sbOk :
  {Γ' Γ : Cx}
  {σ : Sb}
  (_ : Γ' ⊢ˢ σ ∶ Γ)
  → ---------------
  Ok Γ

sbOk (◇ˢ _) = ok◇
sbOk (⨟ˢ q q' _ q'') = ok⨟ q' q'' (sbOk q)

okSb :
  {Γ' Γ : Cx}
  {σ : Sb}
  (_ : Γ' ⊢ˢ σ ∶ Γ)
  → ---------------
  Ok Γ'

okSb (◇ˢ q) = q
okSb (⨟ˢ q _ _ _) = okSb q

sb＝Ok :
  {Γ' Γ : Cx}
  {σ σ' : Sb}
  (_ : Γ' ⊢ˢ σ ＝ σ' ∶ Γ)
  → --------------------
  Ok Γ

sb＝Ok (＝◇ˢ _) = ok◇
sb＝Ok (＝⨟ˢ q q' _ q'') = ok⨟ q' q'' (sb＝Ok q)

okSb＝ :
  {Γ' Γ : Cx}
  {σ σ' : Sb}
  (_ : Γ' ⊢ˢ σ ＝ σ' ∶ Γ)
  → ---------------------
  Ok Γ'

okSb＝ (＝◇ˢ q) = q
okSb＝ (＝⨟ˢ q _ _ _) = okSb＝ q

----------------------------------------------------------------------
-- Context inversion
----------------------------------------------------------------------
[]⁻¹ :
  {l : Lvl}
  {Γ : Cx}
  {A : Ty}
  {x : 𝔸}
  (_ : Ok (Γ ⨟ x ∶ A ⦂ l))
  → --------------------------
  (x # Γ) ∧ (Γ ⊢ A ⦂ l) ∧ Ok Γ

[]⁻¹ (ok⨟ q₀ q₁ h) = (q₁ , q₀ , h)

----------------------------------------------------------------------
-- Version of the rule for context formation without helper hypothesis
----------------------------------------------------------------------
[]⁻ :
  {l : Lvl}
  {Γ : Cx}
  {A : Ty}
  {x : 𝔸}
  (_ : Γ ⊢ A ⦂ l)
  (_ : x # Γ)
  → ----------------
  Ok (Γ ⨟ x ∶ A ⦂ l)

[]⁻ q q' = ok⨟ q q' (⊢ok q)
