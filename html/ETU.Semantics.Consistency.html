{- Consistency of extensional type theory relative to intensional
Martin-Löf Type Theory with inductive-recursive definitions (safe
Agda) -}

consistent : ¬(∃[ a ] (◇ ⊢ a ∶ 𝐄𝐦𝐩 ⦂ 0))

consistent (_ , q) =
  let (t , _) = tot⟦tm⟧' q (⟦𝐄𝐦𝐩⟧ ⟦◇⟧)
  in Øelim (∥ t ∥ tt)
